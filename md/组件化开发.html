注册组件

注册全局组件

在注册全局组件之前，首先创建一个 Vue3 应用程序的根组件

<style>
  .comps-a,
  .comps-b {
    border: 1px solid #999;
    margin: 3px;
  }
</style>
<body>
  <div id="app"></div>
  <template id="my-app">
    <div class="comps-b">
      <input type="text" v-model="message" />
      <h4>{{ message }}</h4>
    </div>
    <div class="comps-a">
      <h4>{{ title }}</h4>
      <h4>{{ desc }}</h4>
      <button @click="btnClick"></button>
    </div>
  </template>
  <script>
    const App = {
      template: "#my-app",
      data() {
        return {
          message: "Hello World",
          title: "我是标题",
          desc: "内容显示区域...",
        };
      },
      computed: {},
      methods: {
        btnClick() {
          console.log("按钮发生单击");
        },
      },
    };
    Vue.createApp(App).mount("#app");
  </script>
</body>

注册组件分成如下两种：

  1. 全局组件：在任何其他的组件中都可以使用的组件

  2. 局部组件：只有在注册的组件中才能使用的组件

注册全局组件的具体步骤如下：

  1. 使用全局创建的 Vue 实例（app）注册全局组件

  2. 调用 app.component 方法，传入组件名称和组件对象，即可注册一个全局组件

<body>
  <div id="app"></div>

  <template id="my-app">
    <div class="comps-b">
      <input type="text" v-model="message" />
      <h4>{{ message }}</h4>
    </div>
    <!-- 使用全局组件 -->
    <component-a></component-a>
  </template>

  <!-- 编写 component-a 全局组件的模板 -->
  <template id="component-a">
    <div class="comps-a">
      <h4>{{ title }}</h4>
      <h4>{{ desc }}</h4>
      <button @click="btnClick">按钮单击</button>
    </div>
  </template>

  <script>
    const App = {
      template: "#my-app",
      data() {
        return {
          message: "Hello World",
        };
      },
      computed: {},
      methods: {},
    };

    const app = Vue.createApp(App);

    /**
     * 参数：全局组件名称、组件对象
     */
    app.component("component-a", {
      template: "#component-a",
      data() {
        return {
          title: "我是标题",
          desc: "内容显示区域...",
        };
      },
      methods: {
        btnClick() {
          console.log("按钮的单击");
        },
      },
    });

    app.mount("#app");
  </script>
</body>

另外，在 Vue3 中也可以注册多个全局组件，

<body>
  <div id="app"></div>

  <template id="my-app">
    <div class="comps-b">
      <input type="text" v-model="message" />
      <h4>{{ message }}</h4>
    </div>
    <!-- 使用全局组件 -->
    <component-a></component-a>
    <component-b></component-b>
  </template>

  <!-- 编写 component-a 全局组件的模板 -->
  <template id="component-a">
    <div class="comps-a">
      <h4>{{ title }}</h4>
      <h4>{{ desc }}</h4>
      <button @click="btnClick">按钮单击</button>
    </div>
  </template>

  <template id="component-b">
    <div class="comps-b">
      <input type="text" v-model="message" />
      <h4>{{ message }}</h4>
    </div>
  </template>

  <script>
    const App = {
      template: "#my-app",
      data() {
        return {
          message: "Hello World",
        };
      },
      computed: {},
      methods: {},
    };

    const app = Vue.createApp(App);

    /**
     * 参数：全局组件名称、组件对象
     */
    app.component("component-a", {
      template: "#component-a",
      data() {
        return {
          title: "我是标题",
          desc: "内容显示区域...",
        };
      },
      methods: {
        btnClick() {
          console.log("按钮的单击");
        },
      },
    });

    app.component("component-b", {
      template: "#component-b",
      data() {
        return {
          message: "Hello World",
        };
      },
    });

    app.mount("#app");
  </script>
</body>

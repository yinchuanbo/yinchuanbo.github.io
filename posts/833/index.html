格式化@内容

// src/services/_format.js
function _formatContent(obj) {
  obj.contentFormat = obj.content;
  obj.contentFormat = obj.contentFormat.replace(
    /@(.+?)\s-\s(\w+?)\b/g,
    (matchStr, nickName, userName) => {
      return `<a href="/profile/${userName}">@${nickName}</a>`;
    }
  );
}
